def <=> (other)
        return nil unless other.is_a?MatrizDensa
        c1=0
        c2=0
        for i in (0...@fil.to_i)
                for j in (0...@col.to_i)
                        if(self.mat[i][j] > other.mat[i][j])
                                c1+=1
                        elsif(self.mat[i][j] < other.mat[i][j])
                                c2+=1
                        end
                end
        end
        (c1)<=>(c2)
end
